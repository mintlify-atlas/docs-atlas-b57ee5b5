---

ewelink-api ships with TypeScript type definitions in `index.d.ts`. These cover the constructor, all method return values, and every nested object. Import the types you need alongside the default export.

```typescript
import eWeLink, { Device, DeviceState, PowerUsage, LoginInfo } from 'ewelink-api';
```

---

## eWeLink class

The main class exported as the default export.

```typescript
class eWeLink {
  constructor(params:
    | { email: string; password: string; }
    | { at: string; }
    | { at: string; apiKey: string; }
  );
  login(): Promise<LoginInfo>;
  openWebSocket(callback: (data: {}) => void): Promise<any>;
  getDevices(): Promise<Device[]>;
  getDevice(deviceId: string): Promise<Device>;
  getDevicePowerState(deviceId: string, channel?: number): Promise<DeviceState>;
  setDevicePowerState(deviceId: string, state?: string, channel?: number): Promise<DeviceState>;
  toggleDevice(deviceId: string, channel?: number): Promise<DeviceState>;
  getDevicePowerUsage(deviceId: string): Promise<PowerUsage>;
  getDeviceCurrentTH(deviceId: string): Promise<TemperatureHumidity>;
  getDeviceCurrentTemperature(deviceId: string): Promise<TemperatureHumidity>;
  getDeviceCurrentHumidity(deviceId: string): Promise<TemperatureHumidity>;
  getDeviceChannelCount(deviceId: string): Promise<SwitchCount>;
  getFirmwareVersion(deviceId: string): Promise<FirmwareVersion>;
}
```

## DeviceState

Returned by `getDevicePowerState`, `setDevicePowerState`, and `toggleDevice`.

```typescript
interface DeviceState {
  status?: string;   // 'ok' on success
  state?: string;    // 'on' or 'off'
  error?: number;    // error code if request failed
  msg?: string;      // error message if request failed
}
```

## PowerUsage

Returned by `getDevicePowerUsage()`.

```typescript
interface PowerUsage {
  status?: string;
  monthly?: number;   // total kWh for current month
  daily?: Daily[];    // per-day breakdown
  error?: number;
  msg?: string;
}

interface Daily {
  day: number;    // day of month
  usage: number;  // kWh consumed
}
```

## LoginInfo

Returned by `getCredentials()` / `login()`.

```typescript
interface LoginInfo {
  at: string;       // access token
  rt: string;       // refresh token
  user: User;
  region: string;
}

interface User {
  _id: string;
  email: string;
  appId: string;
  lang: string;
  online: boolean;
  onlineTime: string;
  ip: string;
  location: string;
  offlineTime: string;
  userStatus: string;
  appInfos: AppInfo[];
  isAccepEmailAd: boolean;
  bindInfos: LoginBindInfos;
  createdAt: string;
  apikey: string;
}

interface AppInfo {
  appVersion: string;
  os: string;
}

interface LoginBindInfos {
  gaction: string[];
}
```
